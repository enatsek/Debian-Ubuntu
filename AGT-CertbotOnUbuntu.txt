#---Specs
# Automation of SSL and certificate renewal for Apache 2 on Ubuntu
# Based on documents on:
Link:https://certbot.eff.org/
# Apache Site configurations without SSL must be ready on your servers
#   Refer to ApacheOnUbuntu Tutorial

#---A. Ubuntu 20.04 LTS
# 1. Enable the universe repository
#  May not be necessary, start with step 2, if it doesn't work return here
sudo apt-get update
sudo apt-get install software-properties-common
sudo add-apt-repository universe
sudo apt-get update
# 2. Install Certbot
sudo apt-get install certbot python3-certbot-apache
# 3. Enable Certbot
sudo certbot --apache
# 4. Test automatic renewal
sudo certbot renew --dry-run
# 5. All set, certificates generated and installed, all apache
#   configurations are made, jobs added to crontab to renew certs.
#   Check crontab:
systemctl list-timers


#---B. Ubuntu 18.04 LTS 
# 1. Add Certbot PPA
#  May not be necessary, start with step 2, if it doesn't work return here
sudo apt-get update
sudo apt-get install software-properties-common
sudo add-apt-repository universe
sudo add-apt-repository ppa:certbot/certbot
sudo apt-get update
# 2. Install Certbot 
sudo apt-get install certbot python3-certbot-apache
# 3. Enable Certbot
sudo certbot --apache
# 4. Test automatic renewal
sudo certbot renew --dry-run
# 5. All set, certificates generated and installed, all apache
#   configurations are made, jobs added to crontab to renew certs.
#   Check crontab:
systemctl list-timers
